export async function updatePhase(req, res) {
    const { id } = req.params;
    const { name, order, points } = req.body;
    try {
        const updatedPhase = await Phase.findByIdAndUpdate(id, { name, order, points}, { new: true });
        if (!updatedPhase) {
            return res.status(404).json({
                status: "failed",
                data: [],
                message: "Fase não encontrada."
            });
        }
        res.status(200).json({
            status: "success",
            data: [updatedPhase],
            message: "Fase atualizada com sucesso!"
        });
    } catch (err) {
        res.status(500).json({
            status: "error",
            code: 500,
            data: [],
            message: "Ocorreu um erro interno no Servidor durante a operação."
        });
    }
    res.end();
}
